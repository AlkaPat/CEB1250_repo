
Academy Staff Type:


Create Table Academic_Staff_Type  (
employee_no INTEGAR PRIMARY KEY,
first_name VARCHAR(50) not null,
last_name VARCHAR(50) not null,
address VARCHAR(50) not null,
email VARCHAR(50) not null,
homepage VARCHAR(50) not null,
reasearch areas VARCHAR(50) not null,
);



course Type:

Create Table course_Type  (
course_id INTEGAR PRIMARY KEY,
course_name VARCHAR(50) not null,
level INTEGAR,
section INTEGAR,
homepage VARCHAR(50) not null,
professor VARCHAR(50) not null,
course VARCHAR(50) not null,
);



Section Type:

Create Table Section_Type  (
section_id INTEGAR PRIMARY KEY,
semester VARCHAR(50) not null,
year DATE,
homepage VARCHAR(50) not null,
professor VARCHAR(50) not null,
Foreign key course references course Type(course_id) 
);

Project Type:

Create Table Project_Type  (
project_id INTEGAR PRIMARY KEY,
project_acronym VARCHAR(50) not null,
project_Name VARCHAR(50) not null,
description VARCHAR(50) not null,
start_date DATE,
end_date DATE,
budget INTEGAR,
funding_agency VARCHAR(50) not null,
);


Assistant Type:

Create Table Assitant_Type  (
thesis_title VARCHAR(50) not null,
thesis_description VARCHAR(50) not null,
advisor VARCHAR(50) not null,
);


Professor Type:

Create Table professor_type  (
status VARCHAR(50) not null,
tenure_date DATE,
no_courses INTEGAR not null,
advises INTEGAR VARCHAR(50),
teaches INTEGAR VARCHAR(50)
);
